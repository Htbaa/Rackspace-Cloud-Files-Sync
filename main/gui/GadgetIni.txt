TProgressBarUpdater.GadgetList = GadgetList
Local progressUpdater:TProgressBarUpdater = New TProgressBarUpdater
TBackup.SetMsgReceiver(progressUpdater)

TBackup.rcf.SetProgressCallback(TProgressBarUpdater.progressFunction, progressUpdater)

'Show setup screen first if there's no configuration set
If TApp.rcfUsername.Length = 0 Or TApp.rcfKey.Length = 0
	Local Gadget1:TGadget, GadgetArray1$[] = ["SetupWindow"] 
	If GadgetList Gadget1:TGadget = GadgetCommander(Show,GadgetArray1,GadgetList:TList)
Else
	Local Gadget2:TGadget, GadgetArray2$[] = ["Window"] 
	If GadgetList Gadget2:TGadget = GadgetCommander(Show,GadgetArray2,GadgetList:TList)
	TApp.SetupRCF()
End If
